--TABELE
CREATE TABLE STATIUNE
(
    id_statiune          NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    tara                 VARCHAR2(100)          NOT NULL,
    nota                 NUMBER(2, 1)           NOT NULL CHECK (nota BETWEEN 0 AND 5),
    numar_partii         NUMBER                 NOT NULL,
    dificultate_statiune VARCHAR2(20)           NOT NULL,
    strat_zapada         NUMBER(5, 2) DEFAULT 0 NOT NULL,
    pret_bilet_ski       NUMBER(8, 2)           NOT NULL /* pret pentru o zi */
);

CREATE TABLE HOTEL
(
    id_hotel    NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    id_statiune NUMBER(13)          NOT NULL,
    nume        VARCHAR2(100)       NOT NULL,
    nota        NUMBER(2, 1)        NOT NULL CHECK (nota BETWEEN 0 AND 5),
    capacitate  NUMBER(5)           NOT NULL,
    mic_dejun   CHAR(1) DEFAULT 'N' NOT NULL,
    CONSTRAINT fk_hotel_statiune FOREIGN KEY (id_statiune) REFERENCES STATIUNE (id_statiune)
);

CREATE TABLE CAMERA
(
    id_camera      NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    id_hotel       NUMBER(13)   NOT NULL,
    nr_camera      VARCHAR2(10) NOT NULL,
    pret_pe_noapte NUMBER(8, 2) NOT NULL,
    CONSTRAINT fk_camera_hotel FOREIGN KEY (id_hotel) REFERENCES HOTEL (id_hotel)
);

CREATE TABLE UTILIZATOR
(
    id_utilizator     NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    email             VARCHAR2(150)        NOT NULL UNIQUE,
    parola            VARCHAR2(255)        NOT NULL,
    nume              VARCHAR2(50)         NOT NULL,
    data_inregistrare DATE DEFAULT SYSDATE NOT NULL
);

CREATE TABLE REZERVARE
(
    id_rezervare  NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    id_utilizator NUMBER(13)                        NOT NULL,
    id_hotel      NUMBER(13)                        NOT NULL,
    stare         VARCHAR2(20) DEFAULT 'confirmata' NOT NULL,
    CONSTRAINT fk_rezervare_user FOREIGN KEY (id_utilizator) REFERENCES UTILIZATOR (id_utilizator),
    CONSTRAINT fk_rezervare_hotel FOREIGN KEY (id_hotel) REFERENCES HOTEL (id_hotel)
);

CREATE TABLE CUMPARA_BILET
(
    id_bilet              NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    id_utilizator         NUMBER(13)   NOT NULL,
    id_statiune           NUMBER(13)   NOT NULL,
    pret                  NUMBER(8, 2) NOT NULL,
    perioada_valabilitate NUMBER       NOT NULL,
    CONSTRAINT fk_bilet_user FOREIGN KEY (id_utilizator) REFERENCES UTILIZATOR (id_utilizator),
    CONSTRAINT fk_bilet_statiune FOREIGN KEY (id_statiune) REFERENCES STATIUNE (id_statiune)
);

CREATE TABLE FURNIZOR
(
    id_furnizor   NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nume_furnizor VARCHAR2(100) NOT NULL,
    tara          VARCHAR2(50),
    contact_email VARCHAR2(150)
);

CREATE TABLE ECHIPAMENT
(
    id_echipament  NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    id_furnizor    NUMBER           NOT NULL, -- Legătura către entitatea independentă
    tip_echipament VARCHAR2(50)     NOT NULL,
    marime         VARCHAR2(10)     NOT NULL,
    pret           NUMBER(8, 2)     NOT NULL,
    stoc           NUMBER DEFAULT 0 NOT NULL,
    CONSTRAINT fk_echip_furnizor FOREIGN KEY (id_furnizor) REFERENCES FURNIZOR (id_furnizor)
);

CREATE TABLE ACHIZITIONEAZA
(
    id_utilizator       NUMBER               NOT NULL,
    id_echipament       NUMBER               NOT NULL,
    data_achizitioneaza DATE DEFAULT SYSDATE NOT NULL,
    pret_platit         NUMBER(8, 2)         NOT NULL,
    CONSTRAINT pk_achizitioneaza PRIMARY KEY (id_utilizator, id_echipament),
    CONSTRAINT fk_achizitioneaza_user FOREIGN KEY (id_utilizator) REFERENCES UTILIZATOR (id_utilizator),
    CONSTRAINT fk_achizitioneaza_echip FOREIGN KEY (id_echipament) REFERENCES ECHIPAMENT (id_echipament)
);

CREATE TABLE ANGAJATI
(
    id_angajat  NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    id_statiune NUMBER(13)   NOT NULL,
    nume        VARCHAR2(50) NOT NULL,
    functie     VARCHAR2(20) NOT NULL,
    CONSTRAINT fk_angajat_statiune FOREIGN KEY (id_statiune) REFERENCES STATIUNE (id_statiune)
);
CREATE TABLE INSTRUCTOR
(
    id_angajat NUMBER PRIMARY KEY,
    CONSTRAINT fk_instructor_angajat FOREIGN KEY (id_angajat) REFERENCES ANGAJATI (id_angajat)
);

CREATE TABLE OPERATOR
(
    id_angajat NUMBER PRIMARY KEY,
    CONSTRAINT fk_operator_angajat FOREIGN KEY (id_angajat) REFERENCES ANGAJATI (id_angajat)
);

CREATE TABLE SALVAMONT
(
    id_angajat NUMBER PRIMARY KEY,
    CONSTRAINT fk_salvamont_angajat FOREIGN KEY (id_angajat) REFERENCES ANGAJATI (id_angajat)
);

--cerinta 6
--Calculati pretul mediu al unei camere pentru fiecare hotel, apoi pentru fiecare statiune si apoi pentru fiecare tara
--Afisati toate mediile pe tara si puneti in categorii: IEFTIN, STANDARD, PREMIUM (care pot fi ajustate)
CREATE OR REPLACE PROCEDURE p_analiza_piata_dinamica IS
    -- 1. TABLOU IMBRICAT
    TYPE t_lista_tari IS TABLE OF VARCHAR2(100);
    v_tari t_lista_tari;

    -- 2. TABLOU INDEXAT
    TYPE t_medii_index IS TABLE OF NUMBER INDEX BY VARCHAR2(100);
    v_colectie_medii t_medii_index;

    -- 3. VECTOR (VARRAY)
    TYPE t_praguri IS VARRAY(2) OF NUMBER;
    v_categorii t_praguri := t_praguri(150, 300);

    v_pret_mediu NUMBER;
    v_cat_nume   VARCHAR2(20);
BEGIN
    SELECT DISTINCT tara BULK COLLECT INTO v_tari FROM STATIUNE;
    FOR i IN 1..v_tari.COUNT LOOP
            SELECT NVL(AVG(c.pret_pe_noapte), 0) INTO v_pret_mediu
            FROM CAMERA c
                     JOIN HOTEL h ON c.id_hotel = h.id_hotel
                     JOIN STATIUNE s ON h.id_statiune = s.id_statiune
            WHERE s.tara = v_tari(i);

            v_colectie_medii(v_tari(i)) := ROUND(v_pret_mediu, 2);
        END LOOP;
    FOR i IN 1..v_tari.COUNT LOOP
            IF v_colectie_medii(v_tari(i)) < v_categorii(1) THEN
                v_cat_nume := 'IEFTIN';
            ELSIF v_colectie_medii(v_tari(i)) < v_categorii(2) THEN
                v_cat_nume := 'STANDARD';
            ELSE
                v_cat_nume := 'PREMIUM';
            END IF;

            DBMS_OUTPUT.PUT_LINE('Tara: ' || RPAD(v_tari(i), 10) ||
                                 ' | Medie: ' || LPAD(v_colectie_medii(v_tari(i)), 6) || '€' ||
                                 ' | Categorie: ' || v_cat_nume);
        END LOOP;

EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Eroare: ' || SQLERRM);
END;
/

BEGIN
    P_ANALIZA_PIATA_DINAMICA();
END;

--cerinta 7
--Pentru statiunile cu zapada mai mare de 15 cm, afisati numele instructorilor disponibili pentru a preda lectii
CREATE OR REPLACE PROCEDURE p_instructori_zapada IS
    CURSOR c_instructori(p_id NUMBER) IS
        SELECT a.nume
        FROM ANGAJATI a
        WHERE a.id_statiune = p_id AND a.functie = 'instructor';

    v_cursor_statiuni SYS_REFCURSOR;
    v_id_statiune     NUMBER;
    v_tara            VARCHAR2(100);
    v_strat_zapada    NUMBER;
    v_nume_ins        VARCHAR2(50);
BEGIN
    DBMS_OUTPUT.PUT_LINE('   INSTRUCTORI (IN STATIUNILE CU ZAPADA SUFICIENTA) ');

    OPEN v_cursor_statiuni FOR 'SELECT id_statiune, tara, strat_zapada FROM STATIUNE WHERE strat_zapada > 15';
    LOOP
        FETCH v_cursor_statiuni INTO v_id_statiune, v_tara, v_strat_zapada;
        EXIT WHEN v_cursor_statiuni%NOTFOUND;

        DBMS_OUTPUT.PUT_LINE('TARA: ' || v_tara || ' (Statiunea ' || v_id_statiune || ') | Zapada: ' || v_strat_zapada || ' cm');

        OPEN c_instructori(v_id_statiune);
        LOOP
            FETCH c_instructori INTO v_nume_ins;
            EXIT WHEN c_instructori%NOTFOUND;

            DBMS_OUTPUT.PUT_LINE('   -> Instructor: ' || v_nume_ins);
        END LOOP;

        IF c_instructori%ROWCOUNT = 0 THEN
            DBMS_OUTPUT.PUT_LINE('Nu sunt instructori alocati in aceasta statiune.');
        END IF;

        CLOSE c_instructori;
    END LOOP;

    CLOSE v_cursor_statiuni;
END;
/

BEGIN
    p_instructori_zapada;
END;
/


--cerinta 8
--Sa se creeze o funcite care primeste ca parametru numele unui utilizator care a cumparat exact un bilet
--si returneaza dificultatea statiunii pentru care a cumparat biletul

CREATE OR REPLACE FUNCTION f_dificultate_statiune_utilizator(p_nume_utilizator VARCHAR2)
    RETURN VARCHAR2 IS
    v_dificultate STATIUNE.dificultate_statiune%TYPE;
BEGIN
    SELECT s.dificultate_statiune
    INTO v_dificultate
    FROM UTILIZATOR u
             JOIN CUMPARA_BILET cb ON u.id_utilizator = cb.id_utilizator
             JOIN STATIUNE s ON cb.id_statiune = s.id_statiune
    WHERE UPPER(u.nume) = UPPER(p_nume_utilizator);

    DBMS_OUTPUT.PUT_LINE(p_nume_utilizator || ' a cumparat un bilet intr-o statiune unde dificultatea era ' || v_dificultate || '');
    RETURN v_dificultate;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Eroare: Utilizatorul ' || p_nume_utilizator || ' nu a fost gasit sau nu are bilete achizitionate.');
        RETURN 'NO_DATA_FOUND';

    WHEN TOO_MANY_ROWS THEN
        DBMS_OUTPUT.PUT_LINE('Eroare: Utilizatorul ' || p_nume_utilizator || ' are achizitii in mai multe statiuni.');
        RETURN 'TOO_MANY_ROWS';

    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Eroare neasteptata: ' || SQLERRM);
        RETURN 'OTHER_ERROR';
END;
/
DECLARE
    v_rezultat VARCHAR2(255);
BEGIN
    --Cazul unde avem sucess
    v_rezultat := f_dificultate_statiune_utilizator('Andrei');
    --Cazul in care functia este aplicata pe un utilizator care nu exista sau nu are bilete achizitionate
    v_rezultat := f_dificultate_statiune_utilizator('NuExista');
    --Cazul in care functia este aplicata pe un  utilizator are achizitii in mai multe statiuni
    v_rezultat := f_dificultate_statiune_utilizator('Elena');
END;
/



--cerinta 9
--Implementati un sistem de filtrare a statiunilor de ski, unde utilizatorii pot alege:
--tara, nivelul de dificultate, pretul maxim de al biletului de schi si pretul  maxim al unei camere.
--si primesc un rezultat cu toate hotelurile si statiunile care satisfac toate filtrele simultan.


CREATE OR REPLACE PROCEDURE p_filtrare_skiresort_pro(
    p_tara_pref        VARCHAR2 DEFAULT NULL,
    p_dif1             VARCHAR2 DEFAULT NULL,
    p_dif2             VARCHAR2 DEFAULT NULL,
    p_dif3             VARCHAR2 DEFAULT NULL,
    p_pret_bilet_max   NUMBER   DEFAULT NULL,
    p_pret_camera_max  NUMBER   DEFAULT NULL
) IS
    e_fara_dificultate    EXCEPTION;
    e_buget_nerealist     EXCEPTION;
    v_gasit               BOOLEAN := FALSE;
BEGIN
    -- Validare input
    IF p_dif1 IS NULL AND p_dif2 IS NULL AND p_dif3 IS NULL THEN
        RAISE e_fara_dificultate;
    END IF;

    IF (p_pret_bilet_max IS NOT NULL AND p_pret_bilet_max < 50) OR
       (p_pret_camera_max IS NOT NULL AND p_pret_camera_max < 50) THEN
        RAISE e_buget_nerealist;
    END IF;
    FOR rec IN (
        SELECT s.tara,
               s.dificultate_statiune,
               h.nume as hotel_nume,
               MIN(c.pret_pe_noapte) as pret_min,
               MAX(c.pret_pe_noapte) as pret_max,
               cb.pret as pret_pass
        FROM STATIUNE s
                 JOIN HOTEL h ON s.id_statiune = h.id_statiune
                 JOIN CAMERA c ON h.id_hotel = c.id_hotel
                 LEFT JOIN CUMPARA_BILET cb ON s.id_statiune = cb.id_statiune
                 LEFT JOIN REZERVARE r ON h.id_hotel = r.id_hotel
        WHERE (p_tara_pref IS NULL OR UPPER(s.tara) = UPPER(p_tara_pref))
          AND (
            LOWER(s.dificultate_statiune) = LOWER(p_dif1) OR
            LOWER(s.dificultate_statiune) = LOWER(p_dif2) OR
            LOWER(s.dificultate_statiune) = LOWER(p_dif3)
            )
          AND (p_pret_bilet_max IS NULL OR NVL(cb.pret, 0) <= p_pret_bilet_max)
          AND (p_pret_camera_max IS NULL OR c.pret_pe_noapte <= p_pret_camera_max)
        GROUP BY s.tara, s.dificultate_statiune, h.nume, cb.pret
        ) LOOP
            v_gasit := TRUE;
            DBMS_OUTPUT.PUT_LINE('Locatie: ' || RPAD(rec.tara, 10) ||
                                 ' | Dif: ' || RPAD(rec.dificultate_statiune, 10) ||
                                 ' | Hotel: ' || RPAD(rec.hotel_nume, 20) ||
                                 ' | Pret pe noapte: ' || LPAD(rec.pret_min, 3) || ' - ' || LPAD(rec.pret_max, 3) || ' RON' ||
                                 ' | Bilet de ski: ' || NVL(TO_CHAR(rec.pret_pass), 'N/A') || ' RON');
        END LOOP;

    IF NOT v_gasit THEN
        DBMS_OUTPUT.PUT_LINE('Rezultat: Nu s-a gasit nicio statiune conform filtrelor tale.');
    END IF;

EXCEPTION
    WHEN e_fara_dificultate THEN
        DBMS_OUTPUT.PUT_LINE('EROARE: Introduceti cel putin un nivel de dificultate.');
    WHEN e_buget_nerealist THEN
        DBMS_OUTPUT.PUT_LINE('EROARE: Bugetul trebuie sa fie mai mare de 50 RON.');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('EROARE: ' || SQLERRM);
END;
/


BEGIN
    DBMS_OUTPUT.PUT_LINE('--- 1: Filtrare de sucess care gaseste rezultate ---');
    p_filtrare_skiresort_pro('Romania', 'Avansat', 'mediu', NULL, 1000, 2000);


    DBMS_OUTPUT.PUT_LINE(CHR(10) || '--- 2: Filtrare de sucess care nu gaseste rezultate ---');
    p_filtrare_skiresort_pro('Romania', 'Avansat', NULL, NULL, 1000, 1000);


    DBMS_OUTPUT.PUT_LINE(CHR(10) || '--- 3: Exceptie Dificultate ---');
    p_filtrare_skiresort_pro('Franta', NULL, NULL, NULL, 500, 500);

    DBMS_OUTPUT.PUT_LINE(CHR(10) || '--- 4: Exceptie Buget ---');
    p_filtrare_skiresort_pro('Italia', 'mediu', NULL, NULL, 5, 200);
END;
/

    AFTER INSERT OR UPDATE OR DELETE ON CUMPARA_BILET
DECLARE
    v_idx   NUMBER;
    v_total NUMBER;
    v_tara  VARCHAR2(100);
BEGIN
    v_idx := pkg_audit_venituri.v_ids.FIRST;

    WHILE v_idx IS NOT NULL LOOP
            SELECT SUM(pret) INTO v_total
            FROM CUMPARA_BILET
            WHERE id_statiune = v_idx;

            SELECT tara INTO v_tara
            FROM STATIUNE
            WHERE id_statiune = v_idx;

            DBMS_OUTPUT.PUT_LINE('>> Statistica pentru ' || v_tara || ' (ID: ' || v_idx || '):');
            DBMS_OUTPUT.PUT_LINE('   Venit total din bilete: ' || NVL(v_total, 0) || ' RON');

            v_idx := pkg_audit_venituri.v_ids.NEXT(v_idx);
        END LOOP;
END;

CREATE OR REPLACE TRIGGER trg_rezervare_stmt_validate
    AFTER INSERT OR UPDATE ON REZERVARE
DECLARE
    v_capacitate_max   NUMBER;
    v_rezervari_active NUMBER;
    v_hotel_id         NUMBER;
BEGIN
    v_hotel_id := pkg_capacitate_audit.v_hotel_list.FIRST;

    WHILE v_hotel_id IS NOT NULL LOOP
            SELECT capacitate INTO v_capacitate_max
            FROM HOTEL
            WHERE id_hotel = v_hotel_id;

            SELECT COUNT(*) INTO v_rezervari_active
            FROM REZERVARE
            WHERE id_hotel = v_hotel_id
              AND LOWER(stare) = 'confirmata';

            IF v_rezervari_active > v_capacitate_max THEN
                RAISE_APPLICATION_ERROR(-20010,
                                        'EROARE CAPACITATE: Hotelul ' || v_hotel_id ||
                                        ' nu poate avea mai mult de ' || v_capacitate_max || ' rezervari confirmate.');
            END IF;

            v_hotel_id := pkg_capacitate_audit.v_hotel_list.NEXT(v_hotel_id);
        END LOOP;
END;

--cerinta 13
--Procesati automat rezervarile pentru clienti. Sistemul trebuie sa verifice daca hotelul are locuri libere,
-- sa calculeze prețul total inmultind pretul pe noapte cu nnumarul de nopti si
-- sa salveze rezervarea in baza de date, oferind la final un raport al operațiunii.

CREATE OR REPLACE PACKAGE pkg_rezervare_simpla AS
    TYPE t_info_rezervare IS RECORD (
                                        id_hotel    NUMBER,
                                        pret_total  NUMBER
                                    );

    TYPE t_lista_audit IS TABLE OF t_info_rezervare INDEX BY BINARY_INTEGER;
    v_audit t_lista_audit;

    FUNCTION verifica_locuri(p_id_hotel NUMBER) RETURN BOOLEAN;
    FUNCTION calculeaza_pret(p_nopti NUMBER) RETURN NUMBER;

    PROCEDURE proceseaza_rezervare(p_id_h NUMBER, p_id_u NUMBER, p_nopti NUMBER);
    PROCEDURE afiseaza_audit;
    PROCEDURE reseteaza_audit;
END pkg_rezervare_simpla;
/

CREATE OR REPLACE PACKAGE BODY pkg_rezervare_simpla AS

    FUNCTION verifica_locuri(p_id_hotel NUMBER) RETURN BOOLEAN IS
        v_cap NUMBER;
        v_occ NUMBER;
    BEGIN
        SELECT capacitate INTO v_cap FROM HOTEL WHERE id_hotel = p_id_hotel;
        SELECT COUNT(*) INTO v_occ FROM REZERVARE WHERE id_hotel = p_id_hotel AND stare = 'confirmata';
        RETURN v_occ < v_cap;
    END;

    FUNCTION calculeaza_pret(p_nopti NUMBER) RETURN NUMBER IS
    BEGIN
        RETURN p_nopti * 200;
    END;

    PROCEDURE reseteaza_audit IS
    BEGIN
        v_audit.DELETE;
    END;

    PROCEDURE proceseaza_rezervare(p_id_h NUMBER, p_id_u NUMBER, p_nopti NUMBER) IS
        v_pret NUMBER;
        v_idx NUMBER := v_audit.COUNT + 1;
    BEGIN
        IF verifica_locuri(p_id_h) THEN
            v_pret := calculeaza_pret(p_nopti);

            INSERT INTO REZERVARE (id_rezervare, id_utilizator, id_hotel, stare)
            VALUES (rezervare_seq.NEXTVAL, p_id_u, p_id_h, 'confirmata');

            v_audit(v_idx).id_hotel := p_id_h;
            v_audit(v_idx).pret_total := v_pret;
            DBMS_OUTPUT.PUT_LINE('Rezervare efectuata cu succes.');
        ELSE
            DBMS_OUTPUT.PUT_LINE('Eroare: Hotel complet.');
        END IF;
    END;

    PROCEDURE afiseaza_audit IS
    BEGIN
        IF v_audit.COUNT = 0 THEN
            DBMS_OUTPUT.PUT_LINE('Auditul este gol.');
        ELSE
            FOR i IN 1..v_audit.COUNT LOOP
                    DBMS_OUTPUT.PUT_LINE('Audit - Hotel: ' || v_audit(i).id_hotel || ' | Pret: ' || v_audit(i).pret_total || ' RON');
                END LOOP;
        END IF;
    END;

END pkg_rezervare_simpla;
/
